lxz = input(“猜猜是什么数字？：”)
guess = int(lxz)
n = 1
while guess != 8 and n < 3:
n += 1
if guess > 8:
lxz = input(“大了哦,再给你一次机会：”)
guess = int(lxz)
else:
lxz = input(“小了哦,再给你一次机会：”)
guess = int(lxz)
if guess == 8:
print(“牛逼啊，老弟！”)
else:
print(“可惜了，没机会了哦”)
print(“Game Over”)

import random
secret = random.randint(1,5)
lxz = input(“猜猜是什么数字？：”)
guess = int(lxz)
n = 1
while guess != secret and n < 3:
n += 1
if guess > secret:
lxz = input(‘大了哦，请重新输入：’)
guess = int(lxz)
else:
lxz = input(‘小了哦，请重新输入：’)
guess = int(lxz)
if guess == secret:
print(“牛逼啊，老弟”)
else:
print(“机会用完了哦”)
print(“Game Over”)
